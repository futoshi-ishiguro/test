    36	#ifndef _CYCLIC_H_
    37	#define _CYCLIC_H_
    38	
    39	#include "queue.h"
    40	#include "time_event.h"
    41	
    42	/*
    43	 *  周期ハンドラ初期化ブロック
    44	 */
    45	typedef struct cyclic_handler_initialization_block {
    46		ATR	cycatr;		/* 周期ハンドラ属性 */
    47		VP_INT	exinf;		/* 周期ハンドラの拡張情報 */
    48		FP	cychdr;		/* 周期ハンドラの起動番地 */
    49		RELTIM	cyctim;		/* 周期ハンドラの起動周期 */
    50		RELTIM	cycphs;		/* 周期ハンドラの起動位相 */
    51	} CYCINIB;
    52	
    53	/*
    54	 *  周期ハンドラ管理ブロック
    55	 */
    56	typedef struct cyclic_handler_control_block {
    57		const CYCINIB *cycinib;	/* 周期ハンドラ初期化ブロックへのポインタ */
    58		BOOL	cycsta;		/* 周期起動ハンドラ動作状態 */
    59		EVTTIM	evttim;		/* 次に周期ハンドラを起動する時刻 */
    60		TMEVTB	tmevtb;		/* タイムイベントブロック */
    61	} CYCCB;
    62	
    63	/*
    64	 *  周期ハンドラ機能の初期化
    65	 */
    66	extern void	cyclic_initialize(void);
    67	
    68	#endif /* _CYCLIC_H_ */
